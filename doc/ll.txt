<prog> -> <body> EOF

<body> -> <definitions> SCOPE <statements> END SCOPE

<definitions> -> <eols> <definition> EOL <eols> <definitions>
<definitions> -> <eols> E

<definition> -> <function_declaration>
<definition> -> <function_definition>

<function_declaration> -> DECLARE <function_header> EOL <eols>
<function_definition> -> <function_header> EOL <eols> <statements> END FUNCTION

<function_header> -> FUNCTION <id> (<function_params>) AS <type>
<function_params> -> E
<function_params> -> <function_param> <function_n_param>
<function_n_param> -> E
<function_n_param> -> , <function_param> <function_n_param>
<function_param> -> <id> AS <type>


<statements> -> E
<statements> -> <statement_single> EOL <eols> <statements>


<variable_declaration> -> DIM <id> AS <type>

<f_statements> -> E
<f_statements> -> <return> EOL <eols> <f_statements>
<f_statements> -> <statement_single> EOL <eols> <f_statements>

<statement_single> -> <id> EQUAL <expression> // přiřazení
<statement_single> -> INPUT <id> // načtení ze vstupu	
<statement_single> -> PRINT <print_expression> <print_expressions> // tisk na výstup, alespoň jeden příkaz, každý se středníkem
<print_expression> -> <expression> SEMICOLON
<print_expressions> -> E
<print_expressions> -> <print_expression> SEMICOLON <print_expressions>

<condition> -> IF <expression> THEN EOL <eols> <statements> <else_section> END IF
<else_section> -> E
<else_section> -> ELSE EOL <eols> <statements>

<return> -> RETURN <expression>
<return> -> E

<do_while> -> DO WHILE <expression> EOL <eols> <statements> LOOP

// Bonusové zadání, for
<for> -> FOR <id> EQUAL <expression> TO <expression> <step>.....?


<id> -> IDENTIFIER
<type> ->
<eols> -> E
<eols> -> EOL <eols>
